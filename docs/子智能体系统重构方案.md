# 子智能体系统重构方案

## 1. 重构目标

对标 Claude Code 的成熟设计，将当前过度复杂的子智能体系统简化为**四层分离架构**：入口层、执行层、生命周期层、通知层。

**核心原则**：
- 砍掉远程子体（WebSocket/protocol/transport）
- 砍掉 DAG 任务调度
- 砍掉审批中心和四维风险评估
- 支持同步/异步两种执行模式
- 支持预定义 Agent 类型 + 通用 Agent
- 支持 Worktree 隔离

```python
class AgentTool(BaseTool):
    """子智能体工具 — 入口层。

    对标 Claude Code 的 AgentTool.tsx。
    负责参数解析、Agent 选择、模式决策、工具池组装、生命周期启动。
    """

    name = "agent"
    description = "启动子智能体执行复杂任务"

    parameters = {
        "prompt": "任务描述",
        "subagent_type": "Agent 类型（可选，默认 general-purpose）",
        "run_in_background": "是否后台执行（可选，默认 True）",
        "isolation": "隔离方式（可选：worktree）",
        "model": "模型覆盖（可选）",
    }

    async def execute(self, **kwargs) -> str:
        # 1. 解析参数
        prompt = kwargs["prompt"]
        agent_type = kwargs.get("subagent_type", "general-purpose")
        run_in_background = kwargs.get("run_in_background", True)
        isolation = kwargs.get("isolation", "")

        # 2. 匹配 Agent 定义
        agent_def = self._resolve_agent(agent_type)

        # 3. 组装工具池
        tools = self._build_tool_pool(agent_def, run_in_background)

        # 4. Worktree 隔离（如需要）
        worktree_info = None
        if isolation == "worktree" or agent_def.isolation == "worktree":
            worktree_info = await create_agent_worktree(task_id)

        # 5. 同步 vs 异步
        if run_in_background:
            # 异步：注册任务 + 后台启动
            task = register_task(...)
            asyncio.create_task(run_async_lifecycle(task, agent_def, tools, ...))
            return f"子智能体已启动（{task.task_id}），后台执行中"
        else:
            # 同步：直接执行并等待结果
            result = await run_agent(agent_def, tools, prompt, ...)
            return result
```

```python
async def run_agent(
    agent_def: AgentDefinition,
    tools: ToolRegistry,
    prompt: str,
    context: SubagentContext,
    budget: TaskBudget,
    worktree_path: str = "",
) -> str:
    """子智能体执行器 — 驱动 LLM 对话循环。

    对标 Claude Code 的 runAgent.ts。
    """
    # 1. 构建隔离上下文
    isolated_ctx = create_subagent_context(context)

    # 2. 构建系统提示
    system_prompt = build_subagent_prompt(agent_def, worktree_path)

    # 3. 驱动 ReActLoop
    loop = ReActLoop(
        system_prompt=system_prompt,
        tools=tools,
        budget=budget,
        context=isolated_ctx,
    )

    result = await loop.run(prompt)

    # 4. 记录 sidechain transcript
    record_sidechain_transcript(context.agent_id, loop.messages)

    return result
```

### 3.7 生命周期层

```python
class AgentLifecycle:
    """异步子智能体生命周期管理。

    对标 Claude Code 的 LocalAgentTask + runAsyncAgentLifecycle。
    """

    def __init__(self, task: Task, notification_queue: NotificationQueue):
        self.task = task
        self.progress = ProgressTracker()
        self._queue = notification_queue

    async def run(self, agent_def, tools, context, budget):
        """驱动异步执行的完整生命周期。"""
        self.task.status = TaskStatus.RUNNING

        try:
            result = await asyncio.wait_for(
                run_agent(agent_def, tools, self.task.goal, context, budget),
                timeout=budget.max_duration_s,
            )
            self.task.status = TaskStatus.COMPLETED
            self.task.result = result
            self._queue.enqueue_notification(self.task, status="completed")

        except asyncio.TimeoutError:
            self.task.status = TaskStatus.FAILED
            self.task.result = "执行超时"
            self._queue.enqueue_notification(self.task, status="failed")

        except asyncio.CancelledError:
            self.task.status = TaskStatus.KILLED
            self._queue.enqueue_notification(self.task, status="killed")

        except Exception as e:
            self.task.status = TaskStatus.FAILED
            self.task.result = str(e)
            self._queue.enqueue_notification(self.task, status="failed")

    def update_progress(self, tool_name: str, tool_input: dict):
        """更新进度追踪。"""
        self.progress.tool_use_count += 1
        activity = {"tool": tool_name, "input": tool_input}
        self.progress.recent_activities.append(activity)
        if len(self.progress.recent_activities) > 5:
            self.progress.recent_activities.pop(0)
```

### 3.8 通知层

```python
class NotificationQueue:
    """统一通知队列。

    对标 Claude Code 的 messageQueueManager。
    子智能体完成后生成 task-notification，注入母体的下一轮对话。
    """

    def __init__(self):
        self._queue: list[TaskNotification] = []
        self._lock = asyncio.Lock()
        self._event = asyncio.Event()  # 通知有新消息

    def enqueue_notification(self, task: Task, status: str):
        notification = TaskNotification(
            task_id=task.task_id,
            agent_type=task.agent_type,
            goal=task.goal,
            status=status,
            result=task.result,
            spawn_tool_call_id=task.spawn_tool_call_id,
        )
        self._queue.append(notification)
        self._event.set()

    async def drain_notifications(self) -> list[TaskNotification]:
        """取出所有待处理通知。在母体 query 的检查点调用。"""
        async with self._lock:
            notifications = self._queue.copy()
            self._queue.clear()
            self._event.clear()
            return notifications

    def build_synthetic_messages(self, notification: TaskNotification) -> list[dict]:
        """将通知转为 synthetic tool_use + tool_result 消息对。

        对标 Claude Code 的 task-notification → attachment → user message 流程。
        但本项目采用 synthetic tool message 方案（已有设计基础）。
        """
        tool_use_msg = {
            "role": "assistant",
            "content": None,
            "tool_calls": [{
                "id": notification.spawn_tool_call_id,
                "type": "function",
                "function": {
                    "name": "subagent_result",
                    "arguments": json.dumps({
                        "task_id": notification.task_id,
                        "agent_type": notification.agent_type,
                        "goal": notification.goal,
                    }),
                },
            }],
        }

        tool_result_msg = {
            "role": "tool",
            "tool_call_id": notification.spawn_tool_call_id,
            "name": "subagent_result",
            "content": json.dumps({
                "status": notification.status,
                "result": notification.result,
                "source": "async_subagent_callback",
            }),
        }

        return [tool_use_msg, tool_result_msg]
```

### 3.9 工具池构建改造

```python
# assembler.py 增加 profile 参数

SUBAGENT_DENY_TOOLS = {"agent"}  # 子体禁止递归派发

ASYNC_AGENT_ALLOWED_TOOLS = {
    "read_file", "write_file", "edit_file",
    "shell", "web_search", "web_fetch",
    "browser", "plan",
}

def build_tool_registry(
    profile: str = "main",          # "main" | "subagent"
    agent_def: AgentDefinition = None,
    is_async: bool = False,
    **kwargs,
) -> ToolRegistry:
    registry = ToolRegistry()

    # 1. 注册基础工具
    _register_base_tools(registry, **kwargs)

    # 2. 主 Agent 专属：注册 AgentTool
    if profile == "main":
        agent_tool = AgentTool(...)
        registry.register(agent_tool)

    # 3. 子体过滤
    if profile == "subagent":
        # 移除递归派发
        for tool_name in SUBAGENT_DENY_TOOLS:
            registry.unregister(tool_name)

    # 4. Agent 定义级过滤
    if agent_def:
        if "*" not in agent_def.tools:
            registry.keep_only(agent_def.tools)
        for name in agent_def.disallowed_tools:
            registry.unregister(name)

    # 5. 异步白名单
    if is_async:
        registry.keep_only(ASYNC_AGENT_ALLOWED_TOOLS)

    return registry
```

### 3.10 权限简化

```python
# tool_policy/engine.py — 简化为两层

class ToolPolicyEngine:
    """两层工具权限策略。

    Layer 1: 全局禁用（系统级，如危险的系统命令）
    Layer 2: 会话/子体级（基于 permission_mode）

    去掉了原来的 PolicyEngineV2 四维风险评估和审批流程。
    """

    GLOBAL_DENY = frozenset({"rm_rf", "format_disk"})

    def check(self, tool_name: str, args: dict, is_subagent: bool = False) -> bool:
        # Layer 1: 全局禁用
        if tool_name in self.GLOBAL_DENY:
            return False

        # Layer 2: 子体额外限制
        if is_subagent and tool_name in SUBAGENT_DENY_TOOLS:
            return False

        return True
```

```python
# kernel.py 中简化的子体结果注入

async def resume_with_subagent_result(
    self,
    session_key: str,
    synthetic_messages: list[dict],
    runtime_instruction: str = "",
) -> OutboundMessage:
    """子体完成后触发母体续写。"""
    session = self.sessions.get_or_create(session_key)
    history = session.get_history()

    # 组装 messages: history + synthetic tool messages
    messages = history + synthetic_messages

    if runtime_instruction:
        messages.append({
            "role": "system",
            "content": runtime_instruction,
        })

    # 触发 LLM 续写
    result = await self._run_orchestrator(messages)

    session.add_message("assistant", result.content)
    self.sessions.save(session)

    return OutboundMessage(content=result.content)
```
